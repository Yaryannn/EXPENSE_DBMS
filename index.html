📐 Table Schema
schema.sql
CREATE TABLE user ( user_id INT PRIMARY KEY AUTO_INCREMENT, username VARCHAR(50) UNIQUE NOT NULL, email VARCHAR(100) UNIQUE NOT NULL, password_hash VARCHAR(255) NOT NULL ); CREATE TABLE expense ( expense_id INT PRIMARY KEY AUTO_INCREMENT, amount FLOAT NOT NULL CHECK (amount > 0), description VARCHAR(255), date DATE NOT NULL, user_id INT NOT NULL, category_id INT NOT NULL, FOREIGN KEY (user_id) REFERENCES user(user_id), FOREIGN KEY (category_id) REFERENCES category(category_id) );

Aggregation
SELECT c.category_name, SUM(e.amount) AS total_spent FROM expense e JOIN category c ON e.category_id = c.category_id GROUP BY c.category_id ORDER BY total_spent DESC;
JOIN
SELECT e.expense_id, e.amount, c.category_name FROM expense e INNER JOIN category c ON e.category_id = c.category_id WHERE e.user_id = 1 ORDER BY e.date DESC;
Trigger
CREATE TRIGGER check_budget_limit AFTER INSERT ON expense FOR EACH ROW BEGIN IF v_spent > v_limit THEN SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Budget exceeded!'; END IF; END;
View
CREATE VIEW expense_summary_view AS SELECT e.expense_id, u.username, e.amount, c.category_name FROM expense e JOIN user u ON e.user_id = u.user_id JOIN category c ON e.category_id = c.category_id;
